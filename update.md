# conda install 模塊名稱
# 更新模塊
# conda update 模塊名稱
# 更新所有模塊
# conda update --all

# conda update conda
# conda update anaconda

PIP 操作命令
pip list # 看目前系統有安裝哪些套件
pip search mysql # 搜尋相關套件
pip install package # 安裝套件
pip uninstall package # 移除套件
pip show --files package # 秀套件檔案列表
pip list --outdated # 列出過期套件
pip install --upgrade package # 升級
